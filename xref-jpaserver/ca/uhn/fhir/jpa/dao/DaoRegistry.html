23  import ca.uhn.fhir.context.FhirContext;
24  import ca.uhn.fhir.context.RuntimeResourceDefinition;
25  import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
26  import org.hl7.fhir.instance.model.api.IBaseResource;
27  import org.springframework.beans.BeansException;
28  import org.springframework.beans.factory.annotation.Autowired;
29  import org.springframework.context.ApplicationContext;
30  import org.springframework.context.ApplicationContextAware;
31  
32  import java.util.HashMap;
33  import java.util.List;
34  import java.util.Map;
35  import java.util.stream.Collectors;
36  
37  public class DaoRegistry implements ApplicationContextAware {
38  	private ApplicationContext myAppCtx;
39  
40  	@Autowired
41  	private FhirContext myCtx;
42  	private volatile Map<String, IFhirResourceDao<?>> myResourceNameToResourceDao;
43  	private volatile IFhirSystemDao<?, ?> mySystemDao;
44  
45  	@Override
46  	public void setApplicationContext(ApplicationContext theApplicationContext) throws BeansException {
47  		myAppCtx = theApplicationContext;
48  	}
49  
50  	public IFhirSystemDao<?, ?> getSystemDao() {
51  		IFhirSystemDao<?, ?> retVal = mySystemDao;
52  		if (retVal == null) {
53  			retVal = myAppCtx.getBean(IFhirSystemDao.class);
54  			mySystemDao = retVal;
55  		}
56  		return retVal;
57  	}
58  
59  	public IFhirResourceDao<?> getResourceDao(String theResourceName) {
60  		IFhirResourceDao<?> retVal = getResourceNameToResourceDao().get(theResourceName);
61  		if (retVal == null) {
62  			List<String> supportedResourceTypes = getResourceNameToResourceDao()
63  				.keySet()
64  				.stream()
65  				.sorted()
66  				.collect(Collectors.toList());
67  			throw new InvalidRequestException("Unable to process request, this server does not know how to handle resources of type " + theResourceName + " - Can handle: " + supportedResourceTypes);
68  		}
69  		return retVal;
70  
71  	}
72  
73  	public <T extends IBaseResource> IFhirResourceDao<T> getResourceDao(Class<T> theResourceType) {
74  		String resourceName = myCtx.getResourceDefinition(theResourceType).getName();
75  		return (IFhirResourceDao<T>) getResourceDao(resourceName);
76  	}
77  
78  	private Map<String, IFhirResourceDao<?>> getResourceNameToResourceDao() {
79  		Map<String, IFhirResourceDao<?>> retVal = myResourceNameToResourceDao;
80  		if (retVal == null || retVal.isEmpty()) {
81  			retVal = new HashMap<>();
82  			Map<String, IFhirResourceDao> resourceDaos = myAppCtx.getBeansOfType(IFhirResourceDao.class);
83  			for (IFhirResourceDao nextResourceDao : resourceDaos.values()) {
84  				RuntimeResourceDefinition nextResourceDef = myCtx.getResourceDefinition(nextResourceDao.getResourceType());
85  				retVal.put(nextResourceDef.getName(), nextResourceDao);
86  			}
87  			myResourceNameToResourceDao = retVal;
88  		}
89  		return retVal;
90  	}

92  }
