125 	public void cancelAllActiveSearches() {
126 		for (BaseTask next : myIdToSearchTask.values()) {
127 			next.requestImmediateAbort();
128 			try {
129 				next.getCompletionLatch().await(30, TimeUnit.SECONDS);
130 			} catch (InterruptedException e) {
131 				ourLog.warn("Failed to wait for completion", e);
132 			}
133 		}

141 	@Transactional(propagation = Propagation.NEVER)
142 	public List<Long> getResources(final String theUuid, int theFrom, int theTo) {
143 		TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
144 		txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
145 
146 		Search search;
147 		StopWatch sw = new StopWatch();
148 		while (true) {
149 
150 			if (myNeverUseLocalSearchForUnitTests == false) {
151 				BaseTask task = myIdToSearchTask.get(theUuid);
152 				if (task != null) {
153 					ourLog.trace("Local search found");
154 					List<Long> resourcePids = task.getResourcePids(theFrom, theTo);
155 					if (resourcePids != null) {
156 						return resourcePids;
157 					}
158 				}
159 			}
160 
161 			search = txTemplate.execute(t -> mySearchDao.findByUuid(theUuid));
162 
163 			if (search == null) {
164 				ourLog.info("Client requested unknown paging ID[{}]", theUuid);
165 				String msg = myContext.getLocalizer().getMessage(PageMethodBinding.class, "unknownSearchId", theUuid);
166 				throw new ResourceGoneException(msg);
167 			}
168 
169 			verifySearchHasntFailedOrThrowInternalErrorException(search);
170 			if (search.getStatus() == SearchStatusEnum.FINISHED) {
171 				ourLog.info("Search entity marked as finished");
172 				break;
173 			}
174 			if (search.getNumFound() >= theTo) {
175 				ourLog.info("Search entity has {} results so far", search.getNumFound());
176 				break;
177 			}
178 
179 			if (sw.getMillis() > myMaxMillisToWaitForRemoteResults) {
180 				ourLog.error("Search {} of type {} for {}{} timed out after {}ms", search.getId(), search.getSearchType(), search.getResourceType(), search.getSearchQueryString(), sw.getMillis());
181 				throw new InternalErrorException("Request timed out after " + sw.getMillis() + "ms");
182 			}
183 
184 			// If the search was saved in "pass complete mode" it's probably time to
185 			// start a new pass
186 			if (search.getStatus() == SearchStatusEnum.PASSCMPLET) {
187 				Optional<Search> newSearch = tryToMarkSearchAsInProgress(search);
188 				if (newSearch.isPresent()) {
189 					search = newSearch.get();
190 					String resourceType = search.getResourceType();
191 					SearchParameterMap params = search.getSearchParameterMap();
192 					SearchContinuationTask task = new SearchContinuationTask(search, myDaoRegistry.getResourceDao(resourceType), params, resourceType);
193 					myIdToSearchTask.put(search.getUuid(), task);
194 					myExecutor.submit(task);
195 				}
196 			}
197 
198 			try {
199 				Thread.sleep(500);
200 			} catch (InterruptedException e) {
201 				// ignore
202 			}
203 		}
204 
205 		final Pageable page = toPage(theFrom, theTo);
206 		if (page == null) {
207 			return Collections.emptyList();
208 		}
209 
210 		final Search foundSearch = search;
211 
212 		ourLog.trace("Loading stored search");
213 		List<Long> retVal = txTemplate.execute(theStatus -> {
214 			final List<Long> resultPids = new ArrayList<>();
215 			Page<Long> searchResultPids = mySearchResultDao.findWithSearchUuid(foundSearch, page);
216 			for (Long next : searchResultPids) {
217 				resultPids.add(next);
218 			}
219 			return resultPids;
220 		});
221 		return retVal;
222 	}

224 	private Optional<Search> tryToMarkSearchAsInProgress(Search theSearch) {
225 		ourLog.trace("Going to try to change search status from {} to {}", theSearch.getStatus(), SearchStatusEnum.LOADING);
226 		try {
227 			TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
228 			txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
229 			txTemplate.afterPropertiesSet();
230 			return txTemplate.execute(t -> {
231 				Optional<Search> searchOpt = mySearchDao.findById(theSearch.getId());
232 				Search search = searchOpt.orElseThrow(IllegalStateException::new);
233 				if (search.getStatus() != SearchStatusEnum.PASSCMPLET) {
234 					throw new IllegalStateException("Can't change to LOADING because state is " + search.getStatus());
235 				}
236 				theSearch.setStatus(SearchStatusEnum.LOADING);
237 				Search newSearch = mySearchDao.save(theSearch);
238 				return Optional.of(newSearch);
239 			});
240 		} catch (Exception e) {
241 			ourLog.warn("Failed to activate search: {}", e.toString());
242 			return Optional.empty();
243 		}
244 	}

254 	@Override
255 	public IBundleProvider registerSearch(final IDao theCallingDao, final SearchParameterMap theParams, String theResourceType, CacheControlDirective theCacheControlDirective) {
256 		StopWatch w = new StopWatch();
257 		final String searchUuid = UUID.randomUUID().toString();
258 
259 		ourLog.debug("Registering new search {}", searchUuid);
260 
261 		Class<? extends IBaseResource> resourceTypeClass = myContext.getResourceDefinition(theResourceType).getImplementingClass();
262 		final ISearchBuilder sb = theCallingDao.newSearchBuilder();
263 		sb.setType(resourceTypeClass, theResourceType);
264 		sb.setFetchSize(mySyncSize);
265 
266 		final Integer loadSynchronousUpTo;
267 		if (theCacheControlDirective != null && theCacheControlDirective.isNoStore()) {
268 			if (theCacheControlDirective.getMaxResults() != null) {
269 				loadSynchronousUpTo = theCacheControlDirective.getMaxResults();
270 				if (loadSynchronousUpTo > myDaoConfig.getCacheControlNoStoreMaxResultsUpperLimit()) {
271 					throw new InvalidRequestException(Constants.HEADER_CACHE_CONTROL + " header " + Constants.CACHE_CONTROL_MAX_RESULTS + " value must not exceed " + myDaoConfig.getCacheControlNoStoreMaxResultsUpperLimit());
272 				}
273 			} else {
274 				loadSynchronousUpTo = 100;
275 			}
276 		} else {
277 			loadSynchronousUpTo = null;
278 		}
279 
280 		if (theParams.isLoadSynchronous() || loadSynchronousUpTo != null) {
281 
282 			ourLog.debug("Search {} is loading in synchronous mode", searchUuid);
283 
284 			// Execute the query and make sure we return distinct results
285 			TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
286 			txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
287 			return txTemplate.execute(t -> {
288 
289 				// Load the results synchronously
290 				final List<Long> pids = new ArrayList<>();
291 
292 				Iterator<Long> resultIter = sb.createQuery(theParams, searchUuid);
293 				while (resultIter.hasNext()) {
294 					pids.add(resultIter.next());
295 					if (loadSynchronousUpTo != null && pids.size() >= loadSynchronousUpTo) {
296 						break;
297 					}
298 					if (theParams.getLoadSynchronousUpTo() != null && pids.size() >= theParams.getLoadSynchronousUpTo()) {
299 						break;
300 					}
301 				}
302 
303 				/*
304 				 * For synchronous queries, we load all the includes right away
305 				 * since we're returning a static bundle with all the results
306 				 * pre-loaded. This is ok because syncronous requests are not
307 				 * expected to be paged
308 				 *
309 				 * On the other hand for async queries we load includes/revincludes
310 				 * individually for pages as we return them to clients
311 				 */
312 				final Set<Long> includedPids = new HashSet<>();
313 				includedPids.addAll(sb.loadIncludes(theCallingDao, myContext, myEntityManager, pids, theParams.getRevIncludes(), true, theParams.getLastUpdated()));
314 				includedPids.addAll(sb.loadIncludes(theCallingDao, myContext, myEntityManager, pids, theParams.getIncludes(), false, theParams.getLastUpdated()));
315 
316 				List<IBaseResource> resources = new ArrayList<>();
317 				sb.loadResourcesByPid(pids, resources, includedPids, false, myEntityManager, myContext, theCallingDao);
318 				return new SimpleBundleProvider(resources);
319 			});
320 		}
321 
322 		/*
323 		 * See if there are any cached searches whose results we can return
324 		 * instead
325 		 */
326 		boolean useCache = true;
327 		if (theCacheControlDirective != null && theCacheControlDirective.isNoCache() == true) {
328 			useCache = false;
329 		}
330 		final String queryString = theParams.toNormalizedQueryString(myContext);
331 		if (theParams.getEverythingMode() == null) {
332 			if (myDaoConfig.getReuseCachedSearchResultsForMillis() != null && useCache) {
333 
334 				final Date createdCutoff = new Date(System.currentTimeMillis() - myDaoConfig.getReuseCachedSearchResultsForMillis());
335 				final String resourceType = theResourceType;
336 
337 				TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
338 				PersistedJpaBundleProvider foundSearchProvider = txTemplate.execute(t -> {
339 					Search searchToUse = null;
340 
341 					int hashCode = queryString.hashCode();
342 					Collection<Search> candidates = mySearchDao.find(resourceType, hashCode, createdCutoff);
343 					for (Search nextCandidateSearch : candidates) {
344 						if (queryString.equals(nextCandidateSearch.getSearchQueryString())) {
345 							searchToUse = nextCandidateSearch;
346 						}
347 					}
348 
349 					PersistedJpaBundleProvider retVal = null;
350 					if (searchToUse != null) {
351 						ourLog.info("Reusing search {} from cache", searchToUse.getUuid());
352 						searchToUse.setSearchLastReturned(new Date());
353 						mySearchDao.updateSearchLastReturned(searchToUse.getId(), new Date());
354 
355 						retVal = new PersistedJpaBundleProvider(searchToUse.getUuid(), theCallingDao);
356 						retVal.setCacheHit(true);
357 
358 						populateBundleProvider(retVal);
359 					}
360 
361 					return retVal;
362 				});
363 
364 				if (foundSearchProvider != null) {
365 					return foundSearchProvider;
366 				}
367 
368 			}
369 		}
370 
371 		Search search = new Search();
372 		populateSearchEntity(theParams, theResourceType, searchUuid, queryString, search);
373 
374 		SearchTask task = new SearchTask(search, theCallingDao, theParams, theResourceType);
375 		myIdToSearchTask.put(search.getUuid(), task);
376 		myExecutor.submit(task);
377 
378 		PersistedJpaSearchFirstPageBundleProvider retVal = new PersistedJpaSearchFirstPageBundleProvider(search, theCallingDao, task, sb, myManagedTxManager);
379 		populateBundleProvider(retVal);
380 
381 		ourLog.debug("Search initial phase completed in {}ms", w.getMillis());
382 		return retVal;
383 
384 	}

443 		private final IDao myCallingDao;
444 		private final String myResourceType;
445 		private final ArrayList<Long> mySyncedPids = new ArrayList<>();
446 		private final CountDownLatch myInitialCollectionLatch = new CountDownLatch(1);

448 		private final ArrayList<Long> myUnsyncedPids = new ArrayList<>();

454 		private Integer myMaxResultsToFetch;
455 		private int myCountFetchedDuringThisPass;
456 		/**

488 		public List<Long> getResourcePids(int theFromIndex, int theToIndex) {
489 			ourLog.info("Requesting search PIDs from {}-{}", theFromIndex, theToIndex);
490 
491 			boolean keepWaiting;
492 			do {
493 				synchronized (mySyncedPids) {
494 					ourLog.trace("Search status is {}", mySearch.getStatus());
495 					keepWaiting = mySyncedPids.size() < theToIndex && mySearch.getStatus() == SearchStatusEnum.LOADING;
496 				}
497 				if (keepWaiting) {
498 					ourLog.info("Waiting, as we only have {} results", mySyncedPids.size());
499 					try {
500 						Thread.sleep(500);
501 					} catch (InterruptedException theE) {
502 						// ignore
503 					}
504 				}
505 			} while (keepWaiting);
506 
507 			ourLog.info("Proceeding, as we have {} results", mySyncedPids.size());
508 
509 			ArrayList<Long> retVal = new ArrayList<>();
510 			synchronized (mySyncedPids) {
511 				verifySearchHasntFailedOrThrowInternalErrorException(mySearch);
512 
513 				int toIndex = theToIndex;
514 				if (mySyncedPids.size() < toIndex) {
515 					toIndex = mySyncedPids.size();
516 				}
517 				for (int i = theFromIndex; i < toIndex; i++) {
518 					retVal.add(mySyncedPids.get(i));
519 				}
520 			}
521 
522 			ourLog.trace("Done syncing results - Wanted {}-{} and returning {} of {}", theFromIndex, theToIndex, retVal.size(), mySyncedPids.size());
523 
524 			return retVal;
525 		}

527 		protected void saveSearch() {
528 			TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
529 			txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
530 			txTemplate.execute(new TransactionCallbackWithoutResult() {
531 				@Override
532 				protected void doInTransactionWithoutResult(TransactionStatus theArg0) {
533 					doSaveSearch();
534 				}
535 
536 			});
537 		}

539 		private void saveUnsynced(final IResultIterator theResultIter) {
540 			TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
541 			txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
542 			txTemplate.execute(new TransactionCallbackWithoutResult() {
543 				@Override
544 				protected void doInTransactionWithoutResult(TransactionStatus theArg0) {
545 					if (mySearch.getId() == null) {
546 						doSaveSearch();
547 					}
548 
549 					List<SearchResult> resultsToSave = Lists.newArrayList();
550 					for (Long nextPid : myUnsyncedPids) {
551 						SearchResult nextResult = new SearchResult(mySearch);
552 						nextResult.setResourcePid(nextPid);
553 						nextResult.setOrder(myCountSaved++);
554 						resultsToSave.add(nextResult);
555 						ourLog.trace("Saving ORDER[{}] Resource {}", nextResult.getOrder(), nextResult.getResourcePid());
556 					}
557 					mySearchResultDao.saveAll(resultsToSave);
558 
559 					synchronized (mySyncedPids) {
560 						int numSyncedThisPass = myUnsyncedPids.size();
561 						ourLog.trace("Syncing {} search results", numSyncedThisPass);
562 						mySyncedPids.addAll(myUnsyncedPids);
563 						myUnsyncedPids.clear();
564 
565 						if (theResultIter.hasNext() == false) {
566 							mySearch.setNumFound(myCountSaved);
567 							int loadedCountThisPass = theResultIter.getSkippedCount() + myCountSaved;
568 							if (myMaxResultsToFetch != null && loadedCountThisPass < myMaxResultsToFetch) {
569 								mySearch.setStatus(SearchStatusEnum.FINISHED);
570 								mySearch.setTotalCount(myCountSaved);
571 							} else if (myAdditionalPrefetchThresholdsRemaining) {
572 								ourLog.trace("Setting search status to PASSCMPLET");
573 								mySearch.setStatus(SearchStatusEnum.PASSCMPLET);
574 								mySearch.setSearchParameterMap(myParams);
575 							} else {
576 								mySearch.setStatus(SearchStatusEnum.FINISHED);
577 								mySearch.setTotalCount(myCountSaved);
578 							}
579 						}
580 					}
581 
582 					mySearch.setNumFound(myCountSaved);
583 
584 					int numSynced;
585 					synchronized (mySyncedPids) {
586 						numSynced = mySyncedPids.size();
587 					}
588 
589 					if (myDaoConfig.getCountSearchResultsUpTo() == null ||
590 						myDaoConfig.getCountSearchResultsUpTo() <= 0 ||
591 						myDaoConfig.getCountSearchResultsUpTo() <= numSynced) {
592 						myInitialCollectionLatch.countDown();
593 					}
594 
595 					doSaveSearch();
596 
597 				}
598 			});
599 
600 		}

621 		/**
622 		 * This is the method which actually performs the search.
623 		 * It is called automatically by the thread pool.
624 		 */
625 		@Override
626 		public Void call() {
627 			StopWatch sw = new StopWatch();
628 
629 			try {
630 				// Create an initial search in the DB and give it an ID
631 				saveSearch();
632 
633 				TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
634 				txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
635 
636 				if (myCustomIsolationSupported) {
637 					txTemplate.setIsolationLevel(TransactionDefinition.ISOLATION_READ_COMMITTED);
638 				}
639 
640 				txTemplate.execute(new TransactionCallbackWithoutResult() {
641 					@Override
642 					protected void doInTransactionWithoutResult(TransactionStatus theStatus) {
643 						doSearch();
644 					}
645 				});
646 
647 				ourLog.info("Completed search for [{}{}] and found {} resources in {}ms", mySearch.getResourceType(), mySearch.getSearchQueryString(), mySyncedPids.size(), sw.getMillis());
648 
649 			} catch (Throwable t) {
650 
651 				/*
652 				 * Don't print a stack trace for client errors (i.e. requests that
653 				 * aren't valid because the client screwed up).. that's just noise
654 				 * in the logs and who needs that.
655 				 */
656 				boolean logged = false;
657 				if (t instanceof BaseServerResponseException) {
658 					BaseServerResponseException exception = (BaseServerResponseException) t;
659 					if (exception.getStatusCode() >= 400 && exception.getStatusCode() < 500) {
660 						logged = true;
661 						ourLog.warn("Failed during search due to invalid request: {}", t.toString());
662 					}
663 				}
664 
665 				if (!logged) {
666 					ourLog.error("Failed during search loading after {}ms", sw.getMillis(), t);
667 				}
668 				myUnsyncedPids.clear();
669 
670 				Throwable rootCause = ExceptionUtils.getRootCause(t);
671 				rootCause = defaultIfNull(rootCause, t);
672 
673 				String failureMessage = rootCause.getMessage();
674 
675 				int failureCode = InternalErrorException.STATUS_CODE;
676 				if (t instanceof BaseServerResponseException) {
677 					failureCode = ((BaseServerResponseException) t).getStatusCode();
678 				}
679 
680 				mySearch.setFailureMessage(failureMessage);
681 				mySearch.setFailureCode(failureCode);
682 				mySearch.setStatus(SearchStatusEnum.FAILED);
683 
684 				saveSearch();
685 
686 			} finally {
687 
688 				myIdToSearchTask.remove(mySearch.getUuid());
689 				myInitialCollectionLatch.countDown();
690 				markComplete();
691 
692 			}
693 			return null;
694 		}

696 		private void doSaveSearch() {
697 
698 			Search newSearch;
699 			if (mySearch.getId() == null) {
700 				newSearch = mySearchDao.save(mySearch);
701 				for (SearchInclude next : mySearch.getIncludes()) {
702 					mySearchIncludeDao.save(next);
703 				}
704 			} else {
705 				newSearch = mySearchDao.save(mySearch);
706 			}
707 
708 			// mySearchDao.save is not supposed to return null, but in unit tests
709 			// it can if the mock search dao isn't set up to handle that
710 			if (newSearch != null) {
711 				mySearch = newSearch;
712 			}
713 		}
714 
715 		/**
716 		 * This method actually creates the database query to perform the
717 		 * search, and starts it.
718 		 */
719 		private void doSearch() {
720 
721 			/*
722 			 * If the user has explicitly requested a _count, perform a
723 			 *
724 			 * SELECT COUNT(*) ....
725 			 *
726 			 * before doing anything else.
727 			 */
728 			boolean wantOnlyCount = SummaryEnum.COUNT.equals(myParams.getSummaryMode());
729 			boolean wantCount = wantOnlyCount || SearchTotalModeEnum.ACCURATE.equals(myParams.getSearchTotalMode());
730 			if (wantCount) {
731 				ourLog.trace("Performing count");
732 				ISearchBuilder sb = newSearchBuilder();
733 				Iterator<Long> countIterator = sb.createCountQuery(myParams, mySearch.getUuid());
734 				Long count = countIterator.next();
735 				ourLog.trace("Got count {}", count);
736 
737 				TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
738 				txTemplate.execute(new TransactionCallbackWithoutResult() {
739 					@Override
740 					protected void doInTransactionWithoutResult(TransactionStatus theArg0) {
741 						mySearch.setTotalCount(count.intValue());
742 						if (wantOnlyCount) {
743 							mySearch.setStatus(SearchStatusEnum.FINISHED);
744 						}
745 						doSaveSearch();
746 						mySearchDao.flush();
747 					}
748 				});
749 				if (wantOnlyCount) {
750 					return;
751 				}
752 			}
753 
754 			ourLog.trace("Done count");
755 			ISearchBuilder sb = newSearchBuilder();
756 
757 			/*
758 			 * Figure out how many results we're actually going to fetch from the
759 			 * database in this pass. This calculation takes into consideration the
760 			 * "pre-fetch thresholds" specified in DaoConfig#getSearchPreFetchThresholds()
761 			 * as well as the value of the _count parameter.
762 			 */
763 			int currentlyLoaded = defaultIfNull(mySearch.getNumFound(), 0);
764 			int minWanted = 0;
765 			if (myParams.getCount() != null) {
766 				minWanted = myParams.getCount();
767 				minWanted = Math.max(minWanted, myPagingProvider.getMaximumPageSize());
768 				minWanted += currentlyLoaded;
769 			}
770 
771 			for (Iterator<Integer> iter = myDaoConfig.getSearchPreFetchThresholds().iterator(); iter.hasNext(); ) {
772 				int next = iter.next();
773 				if (next != -1 && next <= currentlyLoaded) {
774 					continue;
775 				}
776 
777 				if (next == -1) {
778 					sb.setMaxResultsToFetch(null);
779 				} else {
780 					myMaxResultsToFetch = Math.max(next, minWanted);
781 					sb.setMaxResultsToFetch(myMaxResultsToFetch);
782 				}
783 
784 				if (iter.hasNext()) {
785 					myAdditionalPrefetchThresholdsRemaining = true;
786 				}
787 
788 				// If we get here's we've found an appropriate threshold
789 				break;
790 			}
791 
792 			/*
793 			 * Provide any PID we loaded in previous seasrch passes to the
794 			 * SearchBuilder so that we don't get duplicates coming from running
795 			 * the same query again.
796 			 *
797 			 * We could possibly accomplish this in a different way by using sorted
798 			 * results in our SQL query and specifying an offset. I don't actually
799 			 * know if that would be faster or not. At some point should test this
800 			 * idea.
801 			 */
802 			if (myPreviouslyAddedResourcePids != null) {
803 				sb.setPreviouslyAddedResourcePids(myPreviouslyAddedResourcePids);
804 				mySyncedPids.addAll(myPreviouslyAddedResourcePids);
805 			}
806 
807 			/*
808 			 * Construct the SQL query we'll be sending to the database
809 			 */
810 			IResultIterator theResultIterator = sb.createQuery(myParams, mySearch.getUuid());
811 
812 			/*
813 			 * The following loop actually loads the PIDs of the resources
814 			 * matching the search off of the disk and into memory. After
815 			 * every X results, we commit to the HFJ_SEARCH table.
816 			 */
817 			int syncSize = mySyncSize;
818 			while (theResultIterator.hasNext()) {
819 				myUnsyncedPids.add(theResultIterator.next());
820 				myCountFetchedDuringThisPass++;
821 
822 				boolean shouldSync = myUnsyncedPids.size() >= syncSize;
823 
824 				if (myDaoConfig.getCountSearchResultsUpTo() != null &&
825 					myDaoConfig.getCountSearchResultsUpTo() > 0 &&
826 					myDaoConfig.getCountSearchResultsUpTo() < myUnsyncedPids.size()) {
827 					shouldSync = false;
828 				}
829 
830 				if (myUnsyncedPids.size() > 50000) {
831 					shouldSync = true;
832 				}
833 
834 				// If no abort was requested, bail out
835 				Validate.isTrue(isNotAborted(), "Abort has been requested");
836 
837 				if (shouldSync) {
838 					saveUnsynced(theResultIterator);
839 				}
840 
841 				if (myLoadingThrottleForUnitTests != null) {
842 					try {
843 						Thread.sleep(myLoadingThrottleForUnitTests);
844 					} catch (InterruptedException e) {
845 						// ignore
846 					}
847 				}
848 
849 			}
850 
851 			// If no abort was requested, bail out
852 			Validate.isTrue(isNotAborted(), "Abort has been requested");
853 
854 			saveUnsynced(theResultIterator);
855 		}
856 	}

893 	/**
894 	 * A search task is a Callable task that runs in
895 	 * a thread pool to handle an individual search. One instance
896 	 * is created for any requested search and runs from the
897 	 * beginning to the end of the search.
898 	 * <p>
899 	 * Understand:
900 	 * This class executes in its own thread separate from the
901 	 * web server client thread that made the request. We do that
902 	 * so that we can return to the client as soon as possible,
903 	 * but keep the search going in the background (and have
904 	 * the next page of results ready to go when the client asks).
905 	 */
906 	public class SearchTask extends BaseTask {
907 
908 		/**
909 		 * Constructor
910 		 */
911 		public SearchTask(Search theSearch, IDao theCallingDao, SearchParameterMap theParams, String theResourceType) {
912 			super(theSearch, theCallingDao, theParams, theResourceType);
913 		}
914 
915 		/**
916 		 * This method is called by the server HTTP thread, and
917 		 * will block until at least one page of results have been
918 		 * fetched from the DB, and will never block after that.
919 		 */
920 		public Integer awaitInitialSync() {
921 			ourLog.trace("Awaiting initial sync");
922 			do {
923 				try {
924 					if (getInitialCollectionLatch().await(250, TimeUnit.MILLISECONDS)) {
925 						break;
926 					}
927 				} catch (InterruptedException e) {
928 					// Shouldn't happen
929 					throw new InternalErrorException(e);
930 				}
931 			} while (getSearch().getStatus() == SearchStatusEnum.LOADING);
932 			ourLog.trace("Initial sync completed");
933 
934 			return getSearch().getTotalCount();
935 		}
